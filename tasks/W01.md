using namespace std;
 
int main(){
    int L = 1, R = 1000000000, X;
    char input;
    while (input != '=') {
        X = (L + R)/2;
        cout << X << endl;
        cin >> input;
        if ('>' == input)
            L = X + 1;
        else
            R = X;
    }
}
