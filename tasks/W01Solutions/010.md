```c++
#include <iostream>
#include <cmath>
using namespace std;
 
int main() {
    long long n, k, x, y, b, res, floorsDown;
    unsigned short q;
    cin >> n >> k >> x >> y;
    cin >> q;
 
    long long p = (n / k) * x + (n - n / k) * y;
    unsigned long long a[q];
 
    for (int i = 0; i < q; i++) {
        cin >> a[i];
 
        a[i] = (a[i] - 1) % p;
 
        if (x == y) {
            cout << a[i] / y + 1 << endl;
        } else {
            b = x + (k - 1) * y;
 
            floorsDown = (a[i] % b) / y;
            res = a[i] / b * k + min(floorsDown, k - 1) + 1;
 
            cout << res << endl;
        }
    }
}
```
