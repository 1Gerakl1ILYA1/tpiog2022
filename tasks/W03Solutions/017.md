class Solution {
public:
       private: vector<int> v;
    
    public:
    void post(TreeNode* root){
        if(root!=NULL){
            post(root->left);
            post(root->right);
            v.push_back(root->val);
           
        }
    }
    vector<int> postorderTraversal(TreeNode* root) {
     post(root);
        return v;
    }
};
